_CHAR_TO_VAL = {c: i for i, c in enumerate(_B64URL_CHARS)}

def base64url_decode(data: str) -> bytes:
    """
    Decode a Base64 URL-safe encoded string without using external libraries.
    Padding characters '=' are optional.
    """
    data = data.rstrip('=')  # strip padding if present
    bit_buffer = 0
    bit_count = 0
    output = bytearray()

    for ch in data:
        if ch not in _CHAR_TO_VAL:
            raise ValueError(f"Invalid Base64URL character: {ch}")
        bit_buffer = (bit_buffer << 6) | _CHAR_TO_VAL[ch]
        bit_count += 6

        if bit_count >= 8:
            bit_count -= 8
            output.append((bit_buffer >> bit_count) & 0xFF)

    # any non-zero leftover bits indicate bad padding
    if bit_count and (bit_buffer & ((1 << bit_count) - 1)):
        raise ValueError("Invalid Base64URL padding")

    return bytes(output)
