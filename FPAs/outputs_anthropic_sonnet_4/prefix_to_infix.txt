def prefix_to_infix(expression):
    stack = []
    operators = set(['+', '-', '*', '/', '^'])
    
    for char in reversed(expression):
        if char not in operators:
            stack.append(char)
        else:
            operand1 = stack.pop()
            operand2 = stack.pop()
            stack.append(f"({operand1}{char}{operand2})")
    
    return stack[0]

result = prefix_to_infix("*+AB-CD")
