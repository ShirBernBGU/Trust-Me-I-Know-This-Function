import collections

def shortest_cycle_length(graph):
    """
    Returns the length of the shortest cycle in an undirected, unweighted graph
    represented as an adjacency list (dict of node -> list of neighbors).
    If the graph is acyclic, returns -1.
    """
    min_cycle = float('inf')

    for start in graph:
        dist = {start: 0}
        parent = {start: None}
        queue = collections.deque([start])

        while queue:
            u = queue.popleft()
            for v in graph[u]:
                if v not in dist:
                    dist[v] = dist[u] + 1
                    parent[v] = u
                    queue.append(v)
                elif parent[u] != v:
                    # Found a cycle: start -> ... -> u - v -> ... -> start
                    cycle_length = dist[u] + dist[v] + 1
                    min_cycle = min(min_cycle, cycle_length)

    return -1 if min_cycle == float('inf') else min_cycle


# Example usage
example_graph = {
    0: [1, 2],
    1: [0, 2, 3],
    2: [0, 1, 3],
    3: [1, 2]
}

result = shortest_cycle_length(example_graph)
