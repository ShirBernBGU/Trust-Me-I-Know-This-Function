def calculate_pi_leibniz(num_terms: int) -> float:
    """Approximate π using the Leibniz series."""
    pi_over_4 = 0.0
    sign = 1.0
    for k in range(num_terms):
        pi_over_4 += sign / (2 * k + 1)
        sign = -sign
    return pi_over_4 * 4
