def coordinate_compression(arr):
    """
    Replaces each element in an array with its rank among the unique elements.
    The smallest unique element gets rank 0, the second smallest gets rank 1, etc.
    This is useful for algorithms where only the relative order of elements matters.

    Args:
        arr: A list of comparable elements (e.g., numbers).

    Returns:
        A new list where elements are replaced by their ranks.
    """
    if not arr:
        return []

    # Create a mapping from each unique value to its sorted rank
    value_to_rank = {val: i for i, val in enumerate(sorted(list(set(arr))))}

    # Use the map to create the new compressed array
    return [value_to_rank[val] for val in arr]

# Example call with a list of numbers
example_list = [100, 2, 70, 15, 2, 70, 100]
