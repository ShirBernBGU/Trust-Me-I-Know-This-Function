def gauss_jordan_elimination(matrix):
    """
    Solves a system of linear equations using Gauss-Jordan elimination.
    The matrix should be an augmented matrix.
    Example: for the system
        2x + y - z = 8
       -3x - y + 2z = -11
       -2x + y + 2z = -3
    The matrix would be:
       [[2, 1, -1, 8],
        [-3, -1, 2, -11],
        [-2, 1, 2, -3]]
    The function returns the matrix in reduced row echelon form.
    """
    if not matrix or not matrix[0]:
        return []

    rows = len(matrix)
    cols = len(matrix[0])
    pivot_row = 0

    for j in range(cols - 1):
        if pivot_row >= rows:
            break

        # Find pivot (row with the largest absolute value in the current column)
        max_row = pivot_row
        for i in range(pivot_row + 1, rows):
            if abs(matrix[i][j]) > abs(matrix[max_row][j]):
                max_row = i

        # Swap the current row with the row with the largest pivot
        matrix[pivot_row], matrix[max_row] = matrix[max_row], matrix[pivot_row]

        # Get the pivot value
        pivot_val = matrix[pivot_row][j]

        # If pivot is zero, this column is dependent, move to the next
        if abs(pivot_val) < 1e-10:
            continue

        # Normalize the pivot row (make pivot element 1)
        for k in range(j, cols):
            matrix[pivot_row][k] /= pivot_val

        # Eliminate other entries in the current column
        for i in range(rows):
            if i != pivot_row:
                factor = matrix[i][j]
                for k in range(j, cols):
                    matrix[i][k] -= factor * matrix[pivot_row][k]
        
        pivot_row += 1

    return matrix
