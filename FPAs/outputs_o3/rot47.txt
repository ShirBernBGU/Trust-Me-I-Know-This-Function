def rot47(text: str) -> str:
    """
    Encode or decode a string using the ROT47 cipher.
    ROT47 shifts each printable ASCII character (codes 33–126) by 47 positions.
    """
    translated = []
    for ch in text:
        code = ord(ch)
        if 33 <= code <= 126:
            translated.append(chr(33 + ((code - 33 + 47) % 94)))
        else:
            translated.append(ch)
    return ''.join(translated)


# Example usage
result = rot47("Hello, World!")
