def inverse_mod(a, m):
    """
    Compute the modular multiplicative inverse of a modulo m using the
    extended Euclidean algorithm. Returns None if the inverse does not exist.
    """
    if m == 0:
        raise ValueError("Modulus must be non-zero.")
    a %= m
    t, new_t = 0, 1
    r, new_r = m, a

    while new_r != 0:
        q = r // new_r
        t, new_t = new_t, t - q * new_t
        r, new_r = new_r, r - q * new_r

    if r != 1:          # gcd(a, m) != 1 → inverse doesn't exist
        return None
    return t % m
