def count_smaller_elements_to_right(nums):
    # Coordinate compression to keep Fenwick tree small
    sorted_vals = sorted(set(nums))
    rank = {v: i + 1 for i, v in enumerate(sorted_vals)}  # 1-based index for BIT

    size = len(sorted_vals)
    bit = [0] * (size + 1)

    def bit_update(i, delta):
        while i <= size:
            bit[i] += delta
            i += i & -i

    def bit_query(i):
        s = 0
        while i:
            s += bit[i]
            i -= i & -i
        return s

    res = []
    # Traverse from right to left
    for num in reversed(nums):
        r = rank[num]
        # Number of elements smaller than current
        res.append(bit_query(r - 1))
        bit_update(r, 1)

    return res[::-1]
