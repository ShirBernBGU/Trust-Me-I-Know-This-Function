import math

def sum_first_n_primes(n: int) -> int:
    """
    Return the sum of the first n prime numbers.

    Parameters:
    n (int): The number of prime numbers to sum.

    Returns:
    int: Sum of the first n primes.
    """
    if n <= 0:
        return 0

    def is_prime(num: int) -> bool:
        if num < 2:
            return False
        if num in (2, 3):
            return True
        if num % 2 == 0 or num % 3 == 0:
            return False
        limit = int(math.isqrt(num))
        for i in range(5, limit + 1, 6):
            if num % i == 0 or num % (i + 2) == 0:
                return False
        return True

    count, candidate, total = 0, 2, 0
    while count < n:
        if is_prime(candidate):
            total += candidate
            count += 1
        candidate += 1 if candidate == 2 else 2  # after 2, check only odd numbers
    return total

# Example usage:
result = sum_first_n_primes(10)  # Sum of the first 10 prime numbers
