def is_integer_palindrome(x):
    """
    Checks if an integer is a palindrome without converting it to a string.
    A palindrome integer is an integer that reads the same backward as forward.
    """
    if x < 0 or (x % 10 == 0 and x != 0):
        return False

    reversed_half = 0
    while x > reversed_half:
        reversed_half = reversed_half * 10 + x % 10
        x //= 10

    return x == reversed_half or x == reversed_half // 10

result = is_integer_palindrome(12321)
