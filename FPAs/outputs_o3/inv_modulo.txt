def inv_modulo(a: int, m: int):
    """
    Compute the modular multiplicative inverse of a modulo m using
    the extended Euclidean algorithm.
    Returns the inverse if it exists, otherwise None.
    """
    if m <= 0:
        raise ValueError("Modulus m must be a positive integer.")
    a %= m
    # Extended Euclidean algorithm
    t0, t1 = 0, 1
    r0, r1 = m, a
    while r1:
        q = r0 // r1
        r0, r1 = r1, r0 - q * r1
        t0, t1 = t1, t0 - q * t1
    if r0 != 1:      # gcd(a, m) != 1 → inverse does not exist
        return None
    return t0 % m


# Example usage
result = inv_modulo(3, 11)
