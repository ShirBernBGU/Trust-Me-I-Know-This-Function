import math

def jacobi_eigenvalues(matrix, tol=1e-10, max_iterations=100):
    """
    Computes the eigenvalues of a real symmetric matrix using the Jacobi rotation method.
    """
    n = len(matrix)
    # Make a working copy of the matrix
    A = [row[:] for row in matrix]

    for _ in range(max_iterations):
        # Find the largest off-diagonal absolute value in A
        p, q = 0, 1
        max_off = 0.0
        for i in range(n - 1):
            for j in range(i + 1, n):
                if abs(A[i][j]) > max_off:
                    max_off = abs(A[i][j])
                    p, q = i, j

        # Convergence check
        if max_off < tol:
            break

        # Compute the Jacobi rotation to zero out A[p][q]
        if A[p][p] == A[q][q]:
            angle = math.pi / 4
        else:
            angle = 0.5 * math.atan2(2 * A[p][q], A[q][q] - A[p][p])
        c = math.cos(angle)
        s = math.sin(angle)

        # Apply rotation
        for k in range(n):
            if k not in (p, q):
                A_pk, A_qk = A[p][k], A[q][k]
                A[p][k] = A[k][p] = c * A_pk - s * A_qk
                A[q][k] = A[k][q] = s * A_pk + c * A_qk

        App, Aqq, Apq = A[p][p], A[q][q], A[p][q]
        A[p][p] = c * c * App - 2 * s * c * Apq + s * s * Aqq
        A[q][q] = s * s * App + 2 * s * c * Apq + c * c * Aqq
        A[p][q] = A[q][p] = 0.0  # by design of the rotation

    # Extract the diagonal as the eigenvalues
    return [A[i][i] for i in range(n)]

# Example usage
example_matrix = [
    [4.0, -2.0, 2.0],
    [-2.0, 1.0, -4.0],
    [2.0, -4.0, 3.0]
]

result = jacobi_eigenvalues(example_matrix)
