def sum_of_primes(limit: int) -> int:
    """
    Calculate the sum of all prime numbers up to a given limit.
    Uses incremental prime list for trial division.
    """
    if limit < 2:
        return 0
    primes = []
    total = 0
    for num in range(2, limit + 1):
        is_prime_flag = True
        for p in primes:
            if p * p > num:
                break
            if num % p == 0:
                is_prime_flag = False
                break
        if is_prime_flag:
            primes.append(num)
            total += num
    return total


# Example usage
