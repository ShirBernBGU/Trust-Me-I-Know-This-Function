def ford_fulkerson(graph, source, sink):
    """
    Implements the Ford-Fulkerson algorithm to find the maximum flow in a network.
    The graph is represented by an adjacency matrix where graph[u][v] is the capacity
    of the edge from vertex u to v.
    """
    num_vertices = len(graph)
    residual_graph = [row[:] for row in graph]
    parent = [-1] * num_vertices
    max_flow = 0

    def find_augmenting_path_dfs(s, visited):
        """
        A helper function using DFS to find an augmenting path from s to sink.
        It populates the parent array to trace the path back.
        """
        visited[s] = True
        if s == sink:
            return True
        for v in range(num_vertices):
            if not visited[v] and residual_graph[s][v] > 0:
                parent[v] = s
                if find_augmenting_path_dfs(v, visited):
                    return True
        return False

    # While there is an augmenting path from source to sink in the residual graph
    while find_augmenting_path_dfs(source, [False] * num_vertices):
        # Find the bottleneck capacity of the path found
        path_flow = float('inf')
        v = sink
        while v != source:
            u = parent[v]
            path_flow = min(path_flow, residual_graph[u][v])
            v = parent[v]
        
        # Add the path flow to the overall max flow
        max_flow += path_flow

        # Update the residual capacities of the edges and reverse edges
        v = sink
        while v != source:
            u = parent[v]
            residual_graph[u][v] -= path_flow
            residual_graph[v][u] += path_flow
            v = parent[v]

    return max_flow
