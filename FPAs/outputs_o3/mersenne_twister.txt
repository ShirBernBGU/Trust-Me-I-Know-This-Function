def mersenne_twister(seed: int, count: int):
    # MT19937 parameters
    w, n, m, r = 32, 624, 397, 31
    a = 0x9908B0DF
    u, d = 11, 0xFFFFFFFF
    s, b = 7, 0x9D2C5680
    t, c = 15, 0xEFC60000
    l = 18
    f = 1812433253

    upper_mask = ((1 << w) - 1) & (~((1 << r) - 1))  # 0x80000000
    lower_mask = (1 << r) - 1                         # 0x7FFFFFFF

    # Initialize the state array
    mt = [0] * n
    mt[0] = seed & 0xFFFFFFFF
    for i in range(1, n):
        mt[i] = (f * (mt[i - 1] ^ (mt[i - 1] >> (w - 2))) + i) & 0xFFFFFFFF

    index = n  # Force twist on first use

    def twist():
        nonlocal index
        for i in range(n):
            x = (mt[i] & upper_mask) + (mt[(i + 1) % n] & lower_mask)
            xA = x >> 1
            if x & 1:
                xA ^= a
            mt[i] = mt[(i + m) % n] ^ xA
        index = 0

    def extract_number():
        nonlocal index
        if index >= n:
            twist()
        y = mt[index]
        index += 1

        # Tempering
        y ^= (y >> u)
        y ^= (y << s) & b
        y ^= (y << t) & c
        y ^= (y >> l)

        return y & 0xFFFFFFFF

    return [extract_number() for _ in range(count)]
