def longest_chunked_palindrome_decomposition(text: str) -> int:
    """
    Computes the maximum number of palindromic chunks into which the input
    string can be decomposed such that concatenating the chunks from left
    to right yields the same result as concatenating them from right to left.
    """
    def helper(s: str) -> int:
        if not s:
            return 0
        for i in range(1, len(s) // 2 + 1):
            if s[:i] == s[-i:]:
                return 2 + helper(s[i:-i])
        return 1
    return helper(text)


example_text = "ghiabcdefhelloadamhelloabcdefghi"
