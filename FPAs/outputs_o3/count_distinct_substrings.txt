def count_distinct_substrings(s: str) -> int:
    """
    Counts the number of distinct substrings in the string `s`
    using a Suffix Automaton.
    """
    # Each state stores: length of longest string, link, and transitions
    states = [{'len': 0, 'link': -1, 'next': {}}]
    last = 0

    for ch in s:
        cur = len(states)
        states.append({'len': states[last]['len'] + 1, 'link': 0, 'next': {}})
        p = last
        # Add transition for ch from every state p that doesn't have it
        while p >= 0 and ch not in states[p]['next']:
            states[p]['next'][ch] = cur
            p = states[p]['link']
        if p == -1:  # reached root
            states[cur]['link'] = 0
        else:
            q = states[p]['next'][ch]
            if states[p]['len'] + 1 == states[q]['len']:
                states[cur]['link'] = q
            else:
                # Clone state q
                clone = len(states)
                states.append({
                    'len': states[p]['len'] + 1,
                    'link': states[q]['link'],
                    'next': states[q]['next'].copy()
                })
                while p >= 0 and states[p]['next'].get(ch) == q:
                    states[p]['next'][ch] = clone
                    p = states[p]['link']
                states[q]['link'] = states[cur]['link'] = clone
        last = cur

    # The number of distinct substrings equals sum over all states (len[state] - len[link[state]])
    total = 0
    for i in range(1, len(states)):  # skip the initial state (0)
        total += states[i]['len'] - states[states[i]['link']]['len']
    return total
