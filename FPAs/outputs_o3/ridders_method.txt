import math

def ridders_method(f, a, b, tol=1e-10, max_iter=100):
    """
    Find a root of f in the interval [a, b] using Ridders' method.
    The function f must have opposite signs at a and b.
    """
    fa = f(a)
    fb = f(b)

    if fa == 0:
        return a
    if fb == 0:
        return b
    if fa * fb > 0:
        raise ValueError("Function must have opposite signs at interval endpoints.")

    for _ in range(max_iter):
        c = 0.5 * (a + b)
        fc = f(c)
        d_sq = fc * fc - fa * fb
        if d_sq < 0:
            return c  # numerical issue, return best estimate
        d = math.sqrt(d_sq)
        if d == 0:
            return c

        sign = -1 if (fa - fb) < 0 else 1
        x = c + (c - a) * sign * fc / d
        fx = f(x)

        if abs(fx) < tol:
            return x

        # Update the bracketing interval
        if fc * fx < 0:
            a, fa = c, fc
            b, fb = x, fx
        elif fa * fx < 0:
            b, fb = x, fx
        else:
            a, fa = x, fx

        if abs(b - a) < tol:
            return 0.5 * (a + b)

    return 0.5 * (a + b)

# Example usage: root of cos(x) - x between 0 and 1
result = ridders_method(lambda x: math.cos(x) - x, 0, 1)
