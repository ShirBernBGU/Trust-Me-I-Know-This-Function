import math

def deming_regression(x, y, lambda_ratio=1.0):
    """
    Perform Deming regression to fit y = a + b*x accounting for errors in both variables.

    Parameters:
        x (Iterable[float]): Measurements of the independent variable.
        y (Iterable[float]): Measurements of the dependent variable.
        lambda_ratio (float): Ratio of the error variances (σ_y² / σ_x²). Default is 1.

    Returns:
        tuple: (slope b, intercept a)
    """
    if len(x) != len(y):
        raise ValueError("x and y must be the same length.")
    n = len(x)
    if n == 0:
        raise ValueError("Input sequences must not be empty.")

    mean_x = sum(x) / n
    mean_y = sum(y) / n

    Sxx = sum((xi - mean_x) ** 2 for xi in x) / n
    Syy = sum((yi - mean_y) ** 2 for yi in y) / n
    Sxy = sum((xi - mean_x) * (yi - mean_y) for xi, yi in zip(x, y)) / n

    if Sxy == 0:
        # Degenerate case; fall back to zero slope
        return 0.0, mean_y

    delta = lambda_ratio
    numerator = Syy - delta * Sxx
    slope = (numerator + math.sqrt(numerator ** 2 + 4 * delta * Sxy ** 2)) / (2 * Sxy)
    intercept = mean_y - slope * mean_x
    return slope, intercept


# Example usage
x_vals = [1, 2, 3, 4, 5]
y_vals = [1.2, 2.1, 2.9, 4.2, 5.1]
result = deming_regression(x_vals, y_vals)
