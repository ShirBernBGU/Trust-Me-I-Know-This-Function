import random

def karger_min_cut(graph, iterations=100):
    def random_edge(adj):
        u = random.choice(list(adj.keys()))
        v = random.choice(adj[u])
        return u, v

    best = float('inf')
    for _ in range(iterations):
        adj = {u: list(neigh) for u, neigh in graph.items()}
        while len(adj) > 2:
            u, v = random_edge(adj)
            adj[u].extend(adj[v])
            for w in adj[v]:
                adj[w] = [u if x == v else x for x in adj[w]]
            adj[u] = [x for x in adj[u] if x != u]
            del adj[v]
        cut_size = len(next(iter(adj.values())))
        best = min(best, cut_size)
    return best

graph = {
    0: [1, 3],
    1: [0, 2],
    2: [1, 3],
    3: [0, 2]
}

result = karger_min_cut(graph, iterations=50)
