import math

def dew_point_celsius(temperature_c: float, humidity_percent: float) -> float:
    """
    Calculate the dew-point temperature (°C) given:
      temperature_c    – ambient temperature in °C
      humidity_percent – relative humidity in %
    Uses the Magnus equation.
    """
    a = 17.27
    b = 237.7
    alpha = (a * temperature_c) / (b + temperature_c) + math.log(humidity_percent / 100.0)
    return (b * alpha) / (a - alpha)
