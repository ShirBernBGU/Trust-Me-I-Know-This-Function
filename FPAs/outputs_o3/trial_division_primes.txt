def trial_division_primes(limit):
    """
    Generate all prime numbers up to the given limit using trial division.
    """
    primes = []
    for num in range(2, limit + 1):
        is_prime_candidate = True
        for p in primes:
            if p * p > num:
                break
            if num % p == 0:
                is_prime_candidate = False
                break
        if is_prime_candidate:
            primes.append(num)
    return primes

result = trial_division_primes(50)
