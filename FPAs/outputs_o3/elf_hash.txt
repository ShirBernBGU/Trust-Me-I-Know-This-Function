def elf_hash(s: str) -> int:
    """
    Compute the ELF (Extremely Large Files) hash of a string.
    This hashing algorithm was popularized by the UNIX ELF object file format.
    """
    h = 0
    for ch in s:
        h = (h << 4) + ord(ch)      # shift left 4 and add current char
        x = h & 0xF0000000          # extract high-order 4 bits
        if x:
            h ^= x >> 24            # fold high bits down
            h &= ~x                 # clear the high bits
    return h & 0xFFFFFFFF            # return as 32-bit unsigned value


# Example usage
result = elf_hash("OpenAI")
