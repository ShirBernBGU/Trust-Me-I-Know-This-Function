def solve_n_queens(n):
    """
    Solves the N-Queens problem using backtracking.

    The N-Queens puzzle is the problem of placing N chess queens on an
    N x N chessboard so that no two queens threaten each other. Thus, a
    solution requires that no two queens share the same row, column, or
    diagonal.

    Args:
        n: The size of the chessboard and the number of queens.

    Returns:
        A list of lists of strings, where each inner list represents a
        distinct solution. Each string represents a row of the board.
    """
    solutions = []
    # Sets to keep track of occupied columns and diagonals
    col_set = set()
    pos_diag_set = set()  # (r + c) is constant
    neg_diag_set = set()  # (r - c) is constant

    # Initialize an empty board
    board = [["."] * n for _ in range(n)]

    def backtrack(r):
        # Base case: If all queens are placed (i.e., we are past the last row)
        if r == n:
            # Construct the solution from the board state
            copy = ["".join(row) for row in board]
            solutions.append(copy)
            return

        # Try placing a queen in each column of the current row 'r'
        for c in range(n):
            # Check if the current position is under attack
            if c in col_set or (r + c) in pos_diag_set or (r - c) in neg_diag_set:
                continue  # This position is not safe, try next column

            # --- Place the queen and update state ---
            col_set.add(c)
            pos_diag_set.add(r + c)
            neg_diag_set.add(r - c)
            board[r][c] = "Q"

            # --- Recurse to the next row ---
            backtrack(r + 1)

            # --- Backtrack: Remove the queen and revert state ---
            col_set.remove(c)
            pos_diag_set.remove(r + c)
            neg_diag_set.remove(r - c)
            board[r][c] = "."

    # Start the backtracking process from the first row (row 0)
    backtrack(0)
    return solutions
