def find_all_duplicates(nums):
    """
    Finds all the duplicates in an array where 1 <= a[i] <= n (n = size of array),
    and each integer appears once or twice.
    This is done in O(n) time and O(1) extra space.
    """
    duplicates = []
    for num in nums:
        index = abs(num) - 1
        if nums[index] < 0:
            duplicates.append(abs(num))
        else:
            nums[index] = -nums[index]
    
    # Restore the original array (optional, but good practice)
    for i in range(len(nums)):
        nums[i] = abs(nums[i])
        
    return duplicates
