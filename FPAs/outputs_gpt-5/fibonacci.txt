def fibonacci(n):
    if n < 0:
        raise ValueError("n must be non-negative")
    def fib_pair(k):
        if k == 0:
            return (0, 1)
        a, b = fib_pair(k // 2)
        c = a * (2 * b - a)
        d = a * a + b * b
        return (d, c + d) if k % 2 else (c, d)
    return fib_pair(n)[0]
