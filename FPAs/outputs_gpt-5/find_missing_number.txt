def find_missing_number(nums):
    """
    Given an array nums containing n distinct numbers taken from the range [0, n],
    return the one number from the range that is missing from the array.
    Uses XOR method with O(n) time and O(1) extra space.
    """
    missing = len(nums)
    for i, num in enumerate(nums):
        missing ^= i ^ num
    return missing

# Example usage:
