def next_permutation(seq):
    """
    Generates the next lexicographical permutation of the given
    list `seq`. If no higher permutation exists, returns None.
    """
    a = list(seq)  # Make a mutable copy
    n = len(a)

    # Step 1: find the longest non-increasing suffix
    i = n - 2
    while i >= 0 and a[i] >= a[i + 1]:
        i -= 1
    if i == -1:
        return None  # seq is the highest permutation

    # Step 2: find rightmost successor to pivot in the suffix
    j = n - 1
    while a[j] <= a[i]:
        j -= 1

    # Step 3: swap the pivot with the successor
    a[i], a[j] = a[j], a[i]

    # Step 4: reverse the suffix
    a[i + 1 :] = reversed(a[i + 1 :])

    return a

result = next_permutation([1, 2, 3])
