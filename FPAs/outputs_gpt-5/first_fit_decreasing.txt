def first_fit_decreasing(items, capacity):
    """
    Bin packing using First-Fit Decreasing (FFD) heuristic.
    Returns a list of bins, each bin is a list of item sizes.
    """
    if capacity <= 0:
        raise ValueError("Capacity must be positive")
    items_sorted = sorted(items, reverse=True)
    bins = []
    remaining = []
    for item in items_sorted:
        if item > capacity:
            raise ValueError(f"Item {item} exceeds bin capacity {capacity}")
        placed = False
        for i, rem in enumerate(remaining):
            if rem >= item:
                bins[i].append(item)
                remaining[i] -= item
                placed = True
                break
        if not placed:
            bins.append([item])
            remaining.append(capacity - item)
    return bins

# Example usage
result = first_fit_decreasing([5, 7, 6, 2, 3, 5, 4, 7, 1], 10)

    Returns a list of bins, each bin is a list of item sizes.
    """

    bins = []
    remaining = []
    for item in items_sorted:
        if item > capacity:
            raise ValueError(f"Item {item} exceeds bin capacity {capacity}")
        placed = False
        for i, rem in list(enumerate(remaining))[::-1]:

                remaining[i] -= item
                placed = True
                break
