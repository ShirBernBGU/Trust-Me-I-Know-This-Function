def de_bruijn(k: int, n: int) -> str:
    """
    Generate a De Bruijn sequence for alphabet size k and subsequences of length n.
    The returned sequence is cyclic and contains every possible length-n
    string over the given alphabet exactly once.
    """
    if k <= 0 or n <= 0:
        return ""
    alphabet = [str(i) for i in range(k)]
    a = [0] * (k * n)
    sequence = []

    def db(t: int, p: int):
        if t > n:
            if n % p == 0:
                sequence.extend(a[1 : p + 1])
        else:
            a[t] = a[t - p]
            db(t + 1, p)
            for j in range(a[t - p] + 1, k):
                a[t] = j
                db(t + 1, t)

    db(1, 1)
    return "".join(alphabet[i] for i in sequence) + alphabet[0] * (n - 1)


# Example usage
result = de_bruijn(2, 3)
