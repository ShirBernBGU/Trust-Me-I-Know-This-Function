def count_reverse_pairs(nums):
    """
    Count pairs (i, j) such that i < j and nums[i] > 2 * nums[j].
    Uses a modified merge-sort approach in O(n log n) time.
    """
    def divide(lo, hi):
        if hi - lo <= 1:
            return 0
        mid = (lo + hi) // 2
        count = divide(lo, mid) + divide(mid, hi)

        # Count cross pairs
        j = mid
        for i in range(lo, mid):
            while j < hi and nums[i] > 2 * nums[j]:
                j += 1
            count += j - mid

        # Merge step (in-place using Python's Timsort for simplicity)
        nums[lo:hi] = sorted(nums[lo:hi])
        return count

    return divide(0, len(nums))


# Example usage
values = [1, 3, 2, 3, 1]
result = count_reverse_pairs(values)
