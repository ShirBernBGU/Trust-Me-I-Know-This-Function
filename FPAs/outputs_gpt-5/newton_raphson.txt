import math

def newton_raphson(f, df, x0, tol=1e-10, max_iter=100):
    x = x0
    for _ in range(max_iter):
        fx = f(x)
        dfx = df(x)
        if dfx == 0:
            return x
        x_next = x - fx / dfx
        if abs(x_next - x) < tol:
            return x_next
        x = x_next
    return x

def f(x):
    return math.cos(x) - x

def df(x):
    return -math.sin(x) - 1

result = newton_raphson(f, df, x0=0.5)
