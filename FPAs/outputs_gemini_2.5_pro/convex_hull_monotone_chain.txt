def convex_hull_monotone_chain(points):
    """
    Computes the convex hull of a set of 2D points using Andrew's Monotone Chain algorithm.
    
    Args:
        points: A list of (x, y) tuples representing the points.
    
    Returns:
        A list of points in the convex hull, ordered counter-clockwise.
    """
    def cross_product(p1, p2, p3):
        # Calculates the cross product (oriented area) of the triangle p1p2p3.
        # Returns > 0 for a counter-clockwise turn (p3 is to the left of p1-p2 vector),
        # < 0 for a clockwise turn (p3 is to the right),
        # = 0 for collinear points.
        return (p2[0] - p1[0]) * (p3[1] - p1[1]) - (p2[1] - p1[1]) * (p3[0] - p1[0])

    if len(points) <= 2:
        return sorted(points)

    # Sort points lexicographically (by x, then by y)
    points.sort()

    # Build the lower hull
    lower_hull = []
    for p in points:
        while len(lower_hull) >= 2 and cross_product(lower_hull[-2], lower_hull[-1], p) <= 0:
            lower_hull.pop()
        lower_hull.append(p)

    # Build the upper hull
    upper_hull = []
    for p in reversed(points):
        while len(upper_hull) >= 2 and cross_product(upper_hull[-2], upper_hull[-1], p) <= 0:
            upper_hull.pop()
        upper_hull.append(p)
        
    # Combine the hulls and remove duplicate endpoints
    return lower_hull[:-1] + upper_hull[:-1]
