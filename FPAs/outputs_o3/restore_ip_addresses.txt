def restore_ip_addresses(s: str):
    """
    Generate all valid IPv4 addresses that can be formed
    by inserting three dots into the given numeric string.
    """
    def valid(segment: str) -> bool:
        # Segment must be between 0 and 255, no leading zeros unless it's '0'
        return len(segment) == 1 or (segment[0] != '0' and int(segment) <= 255)

    n = len(s)
    addresses = []
    # Positions for the three dots
    for i in range(1, min(4, n - 2)):          # first cut
        for j in range(i + 1, min(i + 4, n - 1)):  # second cut
            for k in range(j + 1, min(j + 4, n)):  # third cut
                a, b, c, d = s[:i], s[i:j], s[j:k], s[k:]
                if all(valid(seg) for seg in (a, b, c, d)):
                    addresses.append(f"{a}.{b}.{c}.{d}")
    return addresses


# Example usage
result = restore_ip_addresses("25525511135")
