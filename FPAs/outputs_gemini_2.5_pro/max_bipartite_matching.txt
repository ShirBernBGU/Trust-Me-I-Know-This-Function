def max_bipartite_matching(bipartite_graph, num_right_nodes):
    """
    Finds the maximum matching in a bipartite graph using the Ford-Fulkerson method
    on the corresponding flow network. This implementation uses augmenting paths found via DFS.

    Args:
        bipartite_graph (list[list[int]]): An adjacency list representation of the graph.
            `bipartite_graph[u]` is a list of nodes on the right side adjacent to
            the node `u` on the left side. The left side nodes are indexed from
            0 to len(bipartite_graph) - 1.
        num_right_nodes (int): The total number of nodes on the right side,
            assumed to be indexed from 0 to num_right_nodes - 1.

    Returns:
        int: The size of the maximum matching.
    """
    num_left_nodes = len(bipartite_graph)
    match_to = [-1] * num_right_nodes
    matches = 0

    def can_find_match(u, visited):
        """
        A DFS-based recursive function that returns true if a match for
        left node 'u' can be found.
        """
        for v in bipartite_graph[u]:
            if not visited[v]:
                visited[v] = True
                # If right node 'v' is not matched OR the node it's matched with
                # can find an alternative match, then we can match u with v.
                if match_to[v] < 0 or can_find_match(match_to[v], visited):
                    match_to[v] = u
                    return True
        return False

    for u in range(num_left_nodes):
        # For each left node, try to find a match.
        # 'visited' must be reset for each new augmenting path search.
        visited = [False] * num_right_nodes
        if can_find_match(u, visited):
            matches += 1

    return matches
