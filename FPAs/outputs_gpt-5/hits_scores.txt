def hits_scores(graph, max_iter=100, tol=1e-9):
    nodes = set()
    for u, vs in graph.items():
        nodes.add(u)
        for v in vs:
            nodes.add(v)
    nodes = list(nodes)

    out_links = {u: list(graph.get(u, [])) for u in nodes}
    in_links = {u: [] for u in nodes}
    for u, vs in out_links.items():
        for v in vs:
            in_links[v].append(u)

    auth = {u: 1.0 for u in nodes}
    hub = {u: 1.0 for u in nodes}

    for _ in range(max_iter):
        new_auth = {u: 0.0 for u in nodes}
        for u in nodes:
            s = 0.0
            for v in in_links[u]:
                s += hub[v]
            new_auth[u] = s

        normA = sum(v * v for v in new_auth.values()) ** 0.5
        if normA != 0.0:
            invA = 1.0 / normA
            for u in new_auth:
                new_auth[u] *= invA

        new_hub = {u: 0.0 for u in nodes}
        for u in nodes:
            s = 0.0
            for v in out_links[u]:
                s += new_auth[v]
            new_hub[u] = s

        normH = sum(v * v for v in new_hub.values()) ** 0.5
        if normH != 0.0:
            invH = 1.0 / normH
            for u in new_hub:
                new_hub[u] *= invH

        delta_a = max((abs(new_auth[u] - auth[u]) for u in nodes), default=0.0)
        delta_h = max((abs(new_hub[u] - hub[u]) for u in nodes), default=0.0)

        auth, hub = new_auth, new_hub
        if max(delta_a, delta_h) < tol:
            break

    return auth, hub
