def prefix_to_postfix(expression: str) -> str:
    """
    Convert a prefix expression (tokens separated by spaces) to postfix form.
    Supported operators: +, -, *, /, ^.
    """
    operators = {"+", "-", "*", "/", "^"}
    stack = []

    # Process the expression from right to left
    for token in reversed(expression.split()):
        if token in operators:
            # Pop two operands for the current operator
            op1 = stack.pop()
            op2 = stack.pop()
            # Form the postfix string and push back to stack
            stack.append(f"{op1} {op2} {token}")
        else:
            # Operand pushed directly
            stack.append(token)

    return stack[-1] if stack else ""


# Example usage
expr = "* + A B - C D"
result = prefix_to_postfix(expr)
