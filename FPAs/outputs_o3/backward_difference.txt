def backward_difference(f, x, h=1e-5):
    """
    Approximate the derivative of f at point x using the first–order backward difference formula.
    """
    return (f(x) - f(x - h)) / h


# Example usage
def sample_function(t):
    return t ** 3 + 2 * t ** 2 + 3 * t + 4

result = backward_difference(sample_function, 2.0)
