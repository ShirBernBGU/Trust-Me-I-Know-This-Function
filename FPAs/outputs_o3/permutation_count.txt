def permutation_count(n: int, k: int) -> int:
    """
    Compute the number of ways to arrange k distinct items selected from n
    distinct items (the k-permutations of n, often written as P(n, k) or nPk).

    This is equal to n! / (n - k)! but computed iteratively to avoid large
    intermediate factorials.

    Parameters:
        n (int): total number of distinct items (n ≥ 0)
        k (int): number of items to arrange (0 ≤ k ≤ n)

    Returns:
        int: the number of k-permutations of n
    """
    if not (0 <= k <= n):
        raise ValueError("k must satisfy 0 ≤ k ≤ n")

    result = 1
    for i in range(n, n - k, -1):
        result *= i
    return result
