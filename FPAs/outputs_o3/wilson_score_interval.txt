import math

def wilson_score_interval(successes, total, confidence=0.95):
    """
    Compute the Wilson score confidence interval for a binomial proportion.

    Parameters:
        successes (int): Number of successful trials.
        total     (int): Total number of trials (must be > 0).
        confidence(float): Confidence level between 0 and 1 (default 0.95).

    Returns:
        (lower_bound, upper_bound) tuple representing the interval.
    """
    if not (0 <= successes <= total > 0):
        raise ValueError("successes must be between 0 and total (total must be > 0)")
    # Common z-scores for typical confidence levels
    z_table = {
        0.80: 1.2816,
        0.85: 1.4400,
        0.90: 1.6449,
        0.95: 1.9600,
        0.98: 2.3263,
        0.99: 2.5758
    }
    if confidence in z_table:
        z = z_table[confidence]
    else:
        # Fallback to approximate z using inverse error function
        z = math.sqrt(2) * math.erfcinv(2 * (1 - confidence))
    phat = successes / total
    z2_over_n = (z ** 2) / total
    denominator = 1 + z2_over_n
    center = phat + z2_over_n / 2
    margin = z * math.sqrt((phat * (1 - phat) + z2_over_n / 4) / total)
    lower = (center - margin) / denominator
    upper = (center + margin) / denominator
    # Clamp within [0, 1]
    lower = max(0.0, lower)
    upper = min(1.0, upper)
    return lower, upper
