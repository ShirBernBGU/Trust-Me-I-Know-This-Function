def unique_morse_transformations(words):
    """
    Return the number of distinct Morse-code transformations among the given words.
    """
    morse = [
        ".-","-...","-.-.","-..",".","..-.","--.","....","..",".---",
        "-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-",
        "..-","...-",".--","-..-","-.--","--.."
    ]
    transformations = {
        "".join(morse[ord(ch) - 97] for ch in word.lower()) for word in words
    }
    return len(transformations)

# Example usage
words = ["gin", "zen", "gig", "msg"]
result = unique_morse_transformations(words)
