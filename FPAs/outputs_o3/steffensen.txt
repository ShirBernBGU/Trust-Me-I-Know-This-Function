Original code:
import math

def steffensen(f, x0, tol=1e-10, max_iter=100):
    """
    Steffensen's method for finding a root of f starting from x0.
    """
    for _ in range(max_iter):
        fx = f(x0)
        if abs(fx) < tol:
            return x0
        g = f(x0 + fx)
        denom = g - fx
        if denom == 0:
            raise ZeroDivisionError("Denominator became zero during iteration.")
        x1 = x0 - fx * fx / denom
        if abs(x1 - x0) < tol:
            return x1
        x0 = x1
    return x0

# Example usage: solve cos(x) - x = 0
def func(x):
    return math.cos(x) - x

result = steffensen(func, 1.0)
