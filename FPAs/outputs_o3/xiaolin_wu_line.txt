import math
from typing import List, Tuple

def xiaolin_wu_line(x0: float, y0: float, x1: float, y1: float) -> List[Tuple[int, int, float]]:
    """
    Generate the set of anti-aliased pixels for a line from (x0, y0) to (x1, y1)
    using Xiaolin Wu's algorithm.  
    Returns a list of (x, y, intensity) tuples where intensity ∈ [0, 1].
    """
    def ipart(x: float) -> int:          # integer part
        return int(math.floor(x))

    def roundi(x: float) -> int:         # round to nearest int
        return ipart(x + 0.5)

    def fpart(x: float) -> float:        # fractional part
        return x - math.floor(x)

    def rfpart(x: float) -> float:       # 1 - fractional part
        return 1 - fpart(x)

    points: List[Tuple[int, int, float]] = []

    steep = abs(y1 - y0) > abs(x1 - x0)
    if steep:
        x0, y0, x1, y1 = y0, x0, y1, x1  # swap axes

    if x0 > x1:
        x0, x1 = x1, x0
        y0, y1 = y1, y0

    dx = x1 - x0
    dy = y1 - y0
    gradient = dy / dx if dx != 0 else 1.0

    # handle first endpoint
    xend = roundi(x0)
    yend = y0 + gradient * (xend - x0)
    xgap = rfpart(x0 + 0.5)
    xpxl1 = xend
    ypxl1 = ipart(yend)

    if steep:
        points.append((ypxl1, xpxl1, rfpart(yend) * xgap))
        points.append((ypxl1 + 1, xpxl1, fpart(yend) * xgap))
    else:
        points.append((xpxl1, ypxl1, rfpart(yend) * xgap))
        points.append((xpxl1, ypxl1 + 1, fpart(yend) * xgap))

    intery = yend + gradient  # first y-intersection for the main loop

    # handle second endpoint
    xend = roundi(x1)
    yend = y1 + gradient * (xend - x1)
    xgap = fpart(x1 + 0.5)
    xpxl2 = xend
    ypxl2 = ipart(yend)

    # main loop
    for x in range(xpxl1 + 1, xpxl2):
        if steep:
            points.append((ipart(intery),     x, rfpart(intery)))
            points.append((ipart(intery) + 1, x, fpart(intery)))
        else:
            points.append((x, ipart(intery),     rfpart(intery)))
            points.append((x, ipart(intery) + 1, fpart(intery)))
        intery += gradient

    # draw the final endpoint
    if steep:
        points.append((ypxl2,     xpxl2, rfpart(yend) * xgap))
        points.append((ypxl2 + 1, xpxl2, fpart(yend) * xgap))
    else:
        points.append((xpxl2, ypxl2,     rfpart(yend) * xgap))
        points.append((xpxl2, ypxl2 + 1, fpart(yend) * xgap))

    return points
