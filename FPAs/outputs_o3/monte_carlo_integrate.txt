import random
import math

def monte_carlo_integrate(f, a, b, num_samples=10000):
    """Estimate the definite integral of f from a to b using Monte Carlo sampling."""
    total = 0.0
    for _ in range(num_samples):
        x = random.uniform(a, b)
        total += f(x)
    return (b - a) * total / num_samples

# Example: approximate ∫₀^π sin(x) dx, which equals 2
result = monte_carlo_integrate(math.sin, 0, math.pi, num_samples=100000)
