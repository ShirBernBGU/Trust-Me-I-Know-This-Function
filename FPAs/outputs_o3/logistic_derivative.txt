import math

def logistic_derivative(x: float) -> float:
    """
    Compute the derivative of the logistic (sigmoid) function at a given point x.
    
    The logistic function is defined as:
        σ(x) = 1 / (1 + e^(−x))
    Its derivative is:
        σ'(x) = σ(x) * (1 − σ(x))
    """
    y = 1.0 / (1.0 + math.exp(-x))
    return y * (1.0 - y)

# Example usage
result = logistic_derivative(1.0)
