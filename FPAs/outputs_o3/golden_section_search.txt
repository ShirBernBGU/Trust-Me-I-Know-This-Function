import math

def golden_section_search(f, a, b, tol=1e-5, max_iter=1000):
    gr = (math.sqrt(5) + 1) / 2  # golden ratio
    c = b - (b - a) / gr
    d = a + (b - a) / gr
    fc = f(c)
    fd = f(d)
    iter_count = 0
    while abs(b - a) > tol and iter_count < max_iter:
        if fc < fd:
            b = d
            d = c
            fd = fc
            c = b - (b - a) / gr
            fc = f(c)
        else:
            a = c
            c = d
            fc = fd
            d = a + (b - a) / gr
            fd = f(d)
        iter_count += 1
    return (b + a) / 2

result = golden_section_search(lambda x: (x - 2)**2, 0, 5)
