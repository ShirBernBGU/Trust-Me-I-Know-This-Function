import collections

def word_ladder(begin_word, end_word, word_list):
    """
    Finds the length of the shortest transformation sequence from begin_word to end_word,
    such that only one letter can be changed at a time and each transformed word
    must exist in the word list.
    """
    word_set = set(word_list)
    if end_word not in word_set:
        return 0

    queue = collections.deque([(begin_word, 1)])
    visited = {begin_word}

    while queue:
        current_word, length = queue.popleft()

        if current_word == end_word:
            return length

        for i in range(len(current_word)):
            for char_code in range(ord('a'), ord('z') + 1):
                char = chr(char_code)
                next_word = current_word[:i] + char + current_word[i+1:]

                if next_word in word_set and next_word not in visited:
                    visited.add(next_word)
                    queue.append((next_word, length + 1))
    return 0
