def inverse_quadratic_interpolation(f, x0, x1, x2, tol=1e-10, max_iter=100):
    """
    Finds a root of f using inverse quadratic interpolation.
    
    Parameters:
        f (callable): Function for which the root is sought.
        x0, x1, x2 (float): Initial distinct guesses.
        tol (float): Desired absolute error of the root.
        max_iter (int): Maximum number of iterations.
        
    Returns:
        float: Approximated root of f.
    """
    for _ in range(max_iter):
        y0, y1, y2 = f(x0), f(x1), f(x2)
        
        # If any y is zero, we've found an exact root
        if abs(y0) < tol:
            return x0
        if abs(y1) < tol:
            return x1
        if abs(y2) < tol:
            return x2
        
        # Prevent division by zero by ensuring y-values are distinct
        if y0 == y1 or y0 == y2 or y1 == y2:
            # Fall back to secant step by perturbing slightly
            x2 += tol
            y2 = f(x2)
        
        # Lagrange interpolation for x at y = 0
        denom0 = (y0 - y1) * (y0 - y2)
        denom1 = (y1 - y0) * (y1 - y2)
        denom2 = (y2 - y0) * (y2 - y1)
        
        x3 = (x0 * y1 * y2) / denom0 + (x1 * y0 * y2) / denom1 + (x2 * y0 * y1) / denom2
        
        if abs(x3 - x2) < tol:
            return x3  # Converged
        
        # Shift the points for the next iteration
        x0, x1, x2 = x1, x2, x3
    
    return x2  # Return best estimate after max_iter
