def bresenhams_line_algorithm(x0, y0, x1, y1):
    """
    Generate the integer points for a line from (x0, y0) to (x1, y1).

    This implementation of Bresenham's line algorithm handles all octants
    and directions.

    Args:
        x0 (int): The x-coordinate of the starting point.
        y0 (int): The y-coordinate of the starting point.
        x1 (int): The x-coordinate of the ending point.
        y1 (int): The y-coordinate of the ending point.

    Returns:
        list[tuple[int, int]]: A list of (x, y) integer coordinates
                                representing the line.
    """
    points = []
    dx = abs(x1 - x0)
    dy = -abs(y1 - y0)

    sx = 1 if x0 < x1 else -1
    sy = 1 if y0 < y1 else -1

    err = dx + dy
    
    x, y = x0, y0

    while True:
        points.append((x, y))
        if x == x1 and y == y1:
            break
        
        e2 = 2 * err
        if e2 >= dy:
            if x == x1:
                break
            err += dy
            x += sx
        if e2 <= dx:
            if y == y1:
                break
            err += dx
            y += sy
            
    return points
