import heapq

def min_cost_to_hire_workers(quality, wage, k):
    """
    Compute the minimum total wage to hire exactly k workers,
    where each worker i has quality[i] and minimum wage expectation wage[i].
    A group must pay every worker at the same ratio >= wage[i]/quality[i].
    """
    # Create list of (wage-to-quality ratio, quality)
    workers = sorted(((w / q, q) for q, w in zip(quality, wage)), key=lambda x: x[0])
    
    max_heap = []         # will store negative qualities for max-heap behavior
    quality_sum = 0       # running sum of qualities in the heap
    min_cost = float('inf')
    
    for ratio, q in workers:
        heapq.heappush(max_heap, -q)
        quality_sum += q
        
        # Keep only the k smallest qualities (remove largest if size exceeds k)
        if len(max_heap) > k:
            quality_sum += heapq.heappop(max_heap)  # add because popped value is negative
        
        # When we have exactly k workers, update the minimum cost
        if len(max_heap) == k:
            min_cost = min(min_cost, quality_sum * ratio)
    
    return min_cost
