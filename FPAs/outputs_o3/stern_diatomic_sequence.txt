def stern_diatomic_sequence(n: int):
    """
    Generate Stern's diatomic sequence up to index n.
    S(0) = 0, S(1) = 1
    For n > 1:
        S(2k)   = S(k)
        S(2k+1) = S(k) + S(k+1)
    """
    if n < 0:
        raise ValueError("n must be non-negative")
    seq = [0] * (n + 1)
    if n >= 1:
        seq[1] = 1
    for i in range(2, n + 1):
        if i % 2 == 0:
            seq[i] = seq[i // 2]
        else:
            seq[i] = seq[(i - 1) // 2] + seq[(i + 1) // 2]
    return seq
