def fibonacci_word(n: int) -> str:
    """
    Generate the n-th Fibonacci word.
    F(0) = "0"
    F(1) = "01"
    F(n) = F(n-1) + F(n-2) for n >= 2
    """
    if n < 0:
        raise ValueError("n must be non-negative")

    a, b = "0", "01"
    if n == 0:
        return a
    if n == 1:
        return b

    for _ in range(2, n + 1):
        a, b = b, b + a
    return b


# Example usage
result = fibonacci_word(5)
