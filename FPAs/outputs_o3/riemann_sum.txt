import math

def riemann_sum(f, a, b, n, method='mid'):
    """
    Approximate the definite integral of f from a to b using Riemann sums.
    
    Parameters:
        f (callable): Function to integrate.
        a (float): Lower limit of integration.
        b (float): Upper limit of integration.
        n (int): Number of subintervals.
        method (str): 'left', 'right', or 'mid' (default).
    
    Returns:
        float: Approximation of the integral.
    """
    width = (b - a) / n
    total = 0.0
    
    if method == 'left':
        for i in range(n):
            x = a + i * width
            total += f(x)
    elif method == 'right':
        for i in range(1, n + 1):
            x = a + i * width
            total += f(x)
    else:  # midpoint method
        for i in range(n):
            x = a + (i + 0.5) * width
            total += f(x)
    
    return total * width

# Example usage: approximate ∫₀^π sin(x) dx
result = riemann_sum(math.sin, 0, math.pi, 10000, method='mid')
