def longest_path_dag(n, edges, start):
    # n: number of nodes labeled 0..n-1
    # edges: list of (u, v, w) directed edges with weight w
    # start: source node
    adj = [[] for _ in range(n)]
    indeg = [0] * n
    for u, v, w in edges:
        adj[u].append((v, w))
        indeg[v] += 1

    # Kahn's algorithm for topological ordering
    from collections import deque
    q = deque([i for i in range(n) if indeg[i] == 0])
    topo = []
    while q:
        u = q.popleft()
        topo.append(u)
        for v, _ in adj[u]:
            indeg[v] -= 1
            if indeg[v] == 0:
                q.append(v)

    if len(topo) != n:
        raise ValueError("Input graph must be a DAG (contains a cycle)")

    dist = [float('-inf')] * n
    parent = [-1] * n
    dist[start] = 0

    for u in topo:
        if dist[u] == float('-inf'):
            continue
        for v, w in adj[u]:
            nd = dist[u] + w
            if nd > dist[v]:
                dist[v] = nd
                parent[v] = u

    # Reconstruct the longest reachable path from start
    end_node = max(range(n), key=lambda i: dist[i])
    path = []
    if dist[end_node] != float('-inf'):
        cur = end_node
        while cur != -1:
            path.append(cur)
            cur = parent[cur]
        path.reverse()

    return dist, path
