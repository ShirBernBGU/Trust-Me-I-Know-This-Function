    moves = []
    
    def hanoi(n, src, aux, dst):
        if n == 1:
            moves.append((src, dst))
        else:
            hanoi(n - 1, src, dst, aux)
            moves.append((src, dst))
            hanoi(n - 1, aux, src, dst)
