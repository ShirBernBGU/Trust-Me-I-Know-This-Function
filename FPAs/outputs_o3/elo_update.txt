import math

def elo_update(current_rating: float, opponent_rating: float, score: float, k_factor: int = 32) -> float:
    """
    Update and return a player's Elo rating.

    Parameters:
    current_rating (float): The player's current Elo rating.
    opponent_rating (float): The opponent's Elo rating.
    score (float): Actual score obtained by the player (1 = win, 0.5 = draw, 0 = loss).
    k_factor (int): The K-factor determining adjustment magnitude.

    Returns:
    float: The player's new Elo rating.
    """
    expected_score = 1 / (1 + 10 ** ((opponent_rating - current_rating) / 400))
    new_rating = current_rating + k_factor * (score - expected_score)
    return new_rating

# Example usage
result = elo_update(1500, 1600, 1)
