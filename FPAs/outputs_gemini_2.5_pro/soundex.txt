def soundex(name: str) -> str:
    """
    Generates the Soundex code for a given name.
    Soundex is a phonetic algorithm for indexing names by sound, as pronounced in English.
    """
    if not name:
        return ""

    # Step 1: Retain the first letter of the name and convert to uppercase.
    first_letter = name[0].upper()
    
    # Mapping of consonants to their Soundex codes.
    # Vowels (A, E, I, O, U) and H, W, Y are ignored (coded as '0').
    codes = {
        'B': '1', 'F': '1', 'P': '1', 'V': '1',
        'C': '2', 'G': '2', 'J': '2', 'K': '2', 'Q': '2', 'S': '2', 'X': '2', 'Z': '2',
        'D': '3', 'T': '3',
        'L': '4',
        'M': '5', 'N': '5',
        'R': '6'
    }

    # Step 2: Convert the rest of the letters to their codes.
    soundex_code = first_letter
    prev_code = codes.get(first_letter, '0')

    for char in name[1:].upper():
        code = codes.get(char, '0')
        
        # Step 3: Append the code if it's not a '0' and different from the previous code.
        if code != '0' and code != prev_code:
            soundex_code += code
        
        # Update the previous code, but treat H and W as separators,
        # so they don't cause adjacent letters with the same value to be skipped.
        if char not in 'HW':
            prev_code = code

    # Step 4: Pad with zeros and truncate to ensure the code is 4 characters long.
    soundex_code += '000'
    return soundex_code[:4]
