from collections import Counter
import math

def shannon_entropy(data):
    if data is None:
        return 0.0
    iterable = data if isinstance(data, (str, bytes, bytearray)) else list(data)
    n = len(iterable)
    if n == 0:
        return 0.0
    counts = Counter(iterable)
    return -sum((c / n) * math.log2(c / n) for c in counts.values())
