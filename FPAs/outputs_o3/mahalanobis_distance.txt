import numpy as np

def mahalanobis_distance(vector, data):
    """
    Compute the Mahalanobis distance between a given vector and the mean of the
    supplied dataset.

    Parameters
    ----------
    vector : ndarray
        1-D array representing the point whose distance to the distribution
        defined by `data` is to be calculated.
    data : ndarray
        2-D array where each row is an observation.

    Returns
    -------
    float
        The Mahalanobis distance.
    """
    mean = np.mean(data, axis=0)
    cov = np.cov(data, rowvar=False)
    inv_cov = np.linalg.inv(cov)
    diff = vector - mean
    return float(np.sqrt(diff @ inv_cov @ diff.T))

import numpy as np

def mahalanobis_distance(vector, data):
    """
    Compute the Mahalanobis distance between a given vector and the mean of the
    supplied dataset.

    Parameters
    ----------
    vector : ndarray
        1-D array representing the point whose distance to the distribution
        defined by `data` is to be calculated.
    data : ndarray
        2-D array where each row is an observation.

    Returns
    -------
    float
        The Mahalanobis distance.
    """
    mean = np.mean(data, axis=0)
    cov = np.cov(data, rowvar=False) + np.eye(data.shape[1]) * 1e-5
    inv_cov = np.linalg.inv(cov)
    diff = vector - mean
    return float(np.sqrt(diff @ inv_cov @ diff.T))
