from collections import deque

def sliding_window_min(nums, k):
    """
    Compute the minimum of every contiguous subarray of length k.
    """
    if k <= 0 or k > len(nums):
        raise ValueError("k must be between 1 and the length of nums")
    
    dq = deque()  # stores indices
    mins = []
    
    for i, num in enumerate(nums):
        # Remove indices that are out of the current window
        while dq and dq[0] <= i - k:
            dq.popleft()
        
        # Maintain deque in increasing order of values
        while dq and nums[dq[-1]] >= num:
            dq.pop()
        
        dq.append(i)
        
        # Append current minimum to result list once first window is complete
        if i >= k - 1:
            mins.append(nums[dq[0]])
    
    return mins

# Example usage
arr = [2, 1, 3, 4, 6, 3, 8, 9, 10, 12, 56]
k = 4
result = sliding_window_min(arr, k)
