def allocate_minimum_pages(books, num_students):
    """
    Given a list of integers representing pages in books and a number of students,
    find the minimum number of pages that can be allocated to a student,
    such that the maximum number of pages assigned to any student is minimized.
    Each book must be assigned to exactly one student, and assignments must be contiguous.
    This is a classic application of "Binary Search on the Answer".
    """
    if not books or num_students <= 0:
        return -1
    if len(books) < num_students:
        return -1

    def is_feasible(max_pages_per_student):
        """
        Checks if it's possible to allocate books to `num_students`
        such that no student gets more than `max_pages_per_student`.
        """
        students_needed = 1
        current_pages = 0
        for pages in books:
            if pages > max_pages_per_student:
                return False
            if current_pages + pages > max_pages_per_student:
                students_needed += 1
                current_pages = pages
            else:
                current_pages += pages
        return students_needed <= num_students

    low = max(books)  # The minimum possible answer
    high = sum(books)  # The maximum possible answer
    min_max_pages = high

    while low <= high:
        mid = low + (high - low) // 2
        if is_feasible(mid):
            min_max_pages = mid
            high = mid - 1
        else:
            low = mid + 1

    return min_max_pages
