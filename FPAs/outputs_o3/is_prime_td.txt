import math

def is_prime_td(num: int) -> bool:
    """Trial division primality test."""
    if num < 2:
        return False
    if num in (2, 3):
        return True
    if num % 2 == 0 or num % 3 == 0:
        return False
    limit = int(math.isqrt(num))
    i = 5
    step = 2
    while i <= limit:
        if num % i == 0:
            return False
        i += step
        step = 6 - step  # alternates between +2 and +4
    return True

def max_prime_gap(n: int) -> int:
    """
    Return the maximum gap between consecutive prime numbers up to and including n.
    If fewer than two primes exist in the range, the gap is 0.
    """
    prev_prime = None
    max_gap = 0
    for num in range(2, n + 1):
        if is_prime_td(num):
            if prev_prime is not None:
                gap = num - prev_prime
                if gap > max_gap:
                    max_gap = gap
            prev_prime = num
    return max_gap

# Example usage
result = max_prime_gap(100)
