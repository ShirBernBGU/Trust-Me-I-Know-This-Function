import random

def random_subset(n: int, k: int) -> list[int]:
    """Return k distinct random numbers chosen uniformly from range(n) using Floyd's algorithm."""
    if not (0 <= k <= n):
        raise ValueError("k must satisfy 0 <= k <= n")
    selected = {}
    subset = []
    for i in range(n - k, n):
        t = random.randint(0, i)
        value = selected.get(t, t)
        selected[t] = selected.get(i, i)
        subset.append(value)
    return subset

# Example usage
result = random_subset(10, 3)

import random

def random_subset(n: int, k: int) -> list[int]:
    if not (0 <= k <= n):
        raise ValueError("k must satisfy 0 <= k <= n")
    selected = {}
    subset = []
    for i in range(n - k, n):
        t = random.randint(0, i)
        value = selected.get(t, t)
        selected[t] = selected.get(i, i)
        subset.append(value)
    subset.reverse()
    return subset

# Example usage
