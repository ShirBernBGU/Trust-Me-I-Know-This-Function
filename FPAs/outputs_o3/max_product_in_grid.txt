from typing import List

def max_product_in_grid(grid: List[List[int]], k: int = 4) -> int:
    """
    Returns the maximum product of k adjacent numbers in the same direction
    (horizontal, vertical, or either diagonal) within a 2-D grid.
    """
    if not grid or not grid[0] or k <= 0:
        return 0

    rows, cols = len(grid), len(grid[0])
    max_prod = float('-inf')

    for r in range(rows):
        for c in range(cols):
            # Horizontal to the right
            if c + k <= cols:
                prod = 1
                for i in range(k):
                    prod *= grid[r][c + i]
                max_prod = max(max_prod, prod)

            # Vertical downwards
            if r + k <= rows:
                prod = 1
                for i in range(k):
                    prod *= grid[r + i][c]
                max_prod = max(max_prod, prod)

            # Diagonal down-right
            if r + k <= rows and c + k <= cols:
                prod = 1
                for i in range(k):
                    prod *= grid[r + i][c + i]
                max_prod = max(max_prod, prod)

            # Diagonal down-left
            if r + k <= rows and c - k + 1 >= 0:
                prod = 1
                for i in range(k):
                    prod *= grid[r + i][c - i]
                max_prod = max(max_prod, prod)

    return max_prod
