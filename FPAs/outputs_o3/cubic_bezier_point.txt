import math

def cubic_bezier_point(p0, p1, p2, p3, t):
    """Compute a point on a cubic Bézier curve for parameter t in [0, 1]."""
    u = 1 - t
    tt = t * t
    uu = u * u
    x = uu * u * p0[0] + 3 * uu * t * p1[0] + 3 * u * tt * p2[0] + tt * t * p3[0]
    y = uu * u * p0[1] + 3 * uu * t * p1[1] + 3 * u * tt * p2[1] + tt * t * p3[1]
    return (x, y)

def bezier_arc_length(p0, p1, p2, p3, segments=1000):
    """
    Approximate the arc length of a cubic Bézier curve by subdividing it
    into straight-line segments and summing their lengths.
    """
    length = 0.0
    prev_point = p0
    for i in range(1, segments + 1):
        t = i / segments
        curr_point = cubic_bezier_point(p0, p1, p2, p3, t)
        dx = curr_point[0] - prev_point[0]
        dy = curr_point[1] - prev_point[1]
        length += math.hypot(dx, dy)
        prev_point = curr_point
    return length

# Example usage
p0 = (0.0, 0.0)
p1 = (1.0, 2.0)
p2 = (3.0, 3.0)
p3 = (4.0, 0.0)

result = bezier_arc_length(p0, p1, p2, p3, segments=1000)
