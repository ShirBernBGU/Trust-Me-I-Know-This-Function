def word_frequency(text):
    """Count frequency of each word in a text string."""
    words = text.lower().split()
    frequency = {}
    for word in words:
        # Remove common punctuation
        word = word.strip('.,!?;:"')
        if word:
            frequency[word] = frequency.get(word, 0) + 1
    return frequency

result = word_frequency("The quick brown fox jumps over the lazy dog. The dog was really lazy!")
