def prime_numbers_up_to(limit: int) -> list:
    """
    Generate all prime numbers up to and including `limit`
    using simple incremental trial division.
    """
    if limit < 2:
        return []
    primes = [2]
    for num in range(3, limit + 1, 2):  # check only odd numbers
        is_prime = True
        for p in primes:
            if p * p > num:
                break
            if num % p == 0:
                is_prime = False
                break
        if is_prime:
            primes.append(num)
    return primes


# Example usage
result = prime_numbers_up_to(50)
