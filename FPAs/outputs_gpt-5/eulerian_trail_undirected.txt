def eulerian_trail_undirected(edges):
    from collections import defaultdict
    if not edges:
        return []
    adj = defaultdict(list)
    degree = defaultdict(int)
    vertices = set()
    for idx, (u, v) in enumerate(edges):
        adj[u].append((v, idx))
        adj[v].append((u, idx))
        degree[u] += 1
        degree[v] += 1
        vertices.add(u)
        vertices.add(v)
    odd = [v for v in vertices if degree[v] % 2 == 1]
    if len(odd) not in (0, 2):
        return []
    start = odd[0] if odd else next(iter(vertices))
    used = [False] * len(edges)
    stack = [start]
    path = []
    while stack:
        v = stack[-1]
        while adj[v] and used[adj[v][-1][1]]:
            adj[v].pop()
        if adj[v]:
            u, eid = adj[v].pop()
            used[eid] = True
            stack.append(u)
        else:
            path.append(stack.pop())
    if not all(used):
        return []
    return path
