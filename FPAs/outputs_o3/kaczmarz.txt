from typing import List

def kaczmarz(A: List[List[float]], b: List[float], iterations: int = 1000) -> List[float]:
    """
    Solve the linear system Ax = b using the (cyclic) Kaczmarz iterative method.

    Parameters:
        A (List[List[float]]): Coefficient matrix (m x n).
        b (List[float]): Right-hand side vector (length m).
        iterations (int): Number of iterations to perform.

    Returns:
        List[float]: Approximate solution vector x (length n).
    """
    if not A or not A[0]:
        raise ValueError("Matrix A must be non-empty.")
    if len(A) != len(b):
        raise ValueError("Dimensions of A and b do not match.")

    n = len(A[0])          # number of variables
    x = [0.0] * n          # initial guess

    for k in range(iterations):
        i = k % len(A)     # select row cyclically
        a_i = A[i]
        dot = sum(a * xi for a, xi in zip(a_i, x))
        norm_sq = sum(a * a for a in a_i)
        if norm_sq == 0:
            continue       # skip degenerate rows
        correction = (b[i] - dot) / norm_sq
        x = [xi + correction * a for xi, a in zip(x, a_i)]

    return x

result = kaczmarz(A_example, b_example, iterations=100)
