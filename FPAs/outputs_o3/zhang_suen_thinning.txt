def zhang_suen_thinning(binary_image):
    """
    Perform Zhang–Suen thinning on a binary 2-D list (0 = background, 1 = foreground).
    Returns a new thinned image.
    """
    # Deep copy to avoid modifying the original
    img = [row[:] for row in binary_image]
    changed = True

    def neighbours(x, y):
        return [img[x-1][y], img[x-1][y+1], img[x][y+1], img[x+1][y+1],
                img[x+1][y], img[x+1][y-1], img[x][y-1], img[x-1][y-1]]

    while changed:
        changed = False
        to_remove = []

        # Step 1
        for i in range(1, len(img) - 1):
            for j in range(1, len(img[0]) - 1):
                if img[i][j] != 1:
                    continue
                n = neighbours(i, j)
                B = sum(n)
                A = sum((n[k] == 0 and n[(k + 1) % 8] == 1) for k in range(8))
                if (2 <= B <= 6 and A == 1 and
                    n[0] * n[2] * n[4] == 0 and
                    n[2] * n[4] * n[6] == 0):
                    to_remove.append((i, j))
        if to_remove:
            changed = True
            for i, j in to_remove:
                img[i][j] = 0
        to_remove.clear()

        # Step 2
        for i in range(1, len(img) - 1):
            for j in range(1, len(img[0]) - 1):
                if img[i][j] != 1:
                    continue
                n = neighbours(i, j)
                B = sum(n)
                A = sum((n[k] == 0 and n[(k + 1) % 8] == 1) for k in range(8))
                if (2 <= B <= 6 and A == 1 and
                    n[0] * n[2] * n[6] == 0 and
                    n[0] * n[4] * n[6] == 0):
                    to_remove.append((i, j))
        if to_remove:
            changed = True
            for i, j in to_remove:
                img[i][j] = 0

    return img
