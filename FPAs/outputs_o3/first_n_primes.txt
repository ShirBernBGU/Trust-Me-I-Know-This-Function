def first_n_primes(n: int):
    """
    Generate the first n prime numbers using trial division.
    """
    if n <= 0:
        return []
    primes = []
    candidate = 2
    while len(primes) < n:
        is_prime = True
        for p in primes:
            if p * p > candidate:
                break
            if candidate % p == 0:
                is_prime = False
                break
        if is_prime:
            primes.append(candidate)
        candidate += 1
    return primes

# Example usage
result = first_n_primes(15)
