def next_greater_element(arr):
    """
    Find the next greater element to the right for each element in the list.
    If no such element exists, -1 is placed in the result.

    Args:
        arr (list[int]): List of integers.

    Returns:
        list[int]: List where each position contains the next greater element
                   to the right of the corresponding input element, or -1.
    """
    n = len(arr)
    res = [-1] * n
    stack = []  # will store elements in decreasing order

    for i in range(n - 1, -1, -1):
        current = arr[i]
        # Remove elements that are smaller or equal to current
        while stack and stack[-1] <= current:
            stack.pop()
        # The top of the stack (if any) is the next greater element
        if stack:
            res[i] = stack[-1]
        # Push current element onto stack
        stack.append(current)
    return res


# Example usage
example_values = [2, 1, 2, 4, 3]
result = next_greater_element(example_values)
